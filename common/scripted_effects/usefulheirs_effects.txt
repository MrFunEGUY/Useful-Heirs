# Heir Setup Non - Adoptive Ddnasties
heir_setup_non_ad = {
	if = {
		limit = {
			has_global_flag = civil_wars_active
			exists = heir
		}
		heir = {
			save_event_target_as = disinherited_heir
		}
		set_country_flag = dissed_heir
	}
	else_if = {
		limit = { exists = heir }
		heir = {
			set_leader_flag = dead_ruler
			kill_leader = { show_notification = no }
		}
	}
	set_heir = event_target:new_chosen_heir
	country_event = { id = usefulheirs.3500 }
}

# Get Governor Leaders
get_governors_heir = {
	set_variable = { which = total_governors value = 0 }

	every_owned_leader = {
		limit = {
			leader_class = governor
			NOR = {
				# has_leader_flag = clone_heir_base
				is_same_value = heir
			}
		}
		set_leader_flag = heir_maybe
		root = { change_variable = { which = total_governors value = 1 } }
	}

	if = {
		limit = { check_variable = { which = total_governors value >= 1 } }
		random_owned_leader = {
			limit = { has_leader_flag = heir_maybe }
			remove_leader_flag = heir_maybe
			save_event_target_as = heir_maybe_1
		}
	}
	if = {
		limit = { check_variable = { which = total_governors value >= 2 } }
		random_owned_leader = {

get_heir_leader_clone_effect = {

	save_event_target_as = new_cloned_heir
	if = {
		limit = {
			owner = {
				is_ai = no
				has_country_flag = heir_name_color_activated
			}
		}
		set_name = {
			key = heir_name_bold
			variable_string = "\\[event_target:heir_leader.GetName]"
		}
	}
}

get_heir_leader_switch_effect = {

	last_created_leader = {
		if = {
			limit = { event_target:heir_leader = { has_leader_flag = appointed_heir } }
			set_leader_flag = appointed_heir
		}
		if = {
			limit = { event_target:heir_leader = { has_leader_flag = adopted_dynastic_name } }
			set_leader_flag = adopted_dynastic_name
		}
	}

	set_heir = event_target:new_cloned_heir

	event_target:heir_leader = {
		kill_leader = { show_notification = no }
	}
}
